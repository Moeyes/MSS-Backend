title V4 - Athletics Management System

// OGS or Provinces
Organization [icon: building, color: blue] {
  id Int pk
  name VARCHAR(100)
  createdAt TIMESTAMP
  leaderID Int fk
  athleticsID Int fk
}

Users [icon: users, color: orange] {
  UserID Int pk
  username VARCHAR(20)
  password VARCHAR(255)
  userLevel Int
  createdAt TIMESTAMP
}

// Register any Leader and Althletics by USERS
Enroll [icon: user-plus, color: red] {
  id Int pk
  name VARCHAR(100)
  gender VARCHAR(10)
  nationality VARCHAR(25)
  dob DATE
  idCard VARCHAR(25)
  address VARCHAR(150)
  createdAt TIMESTAMP
  userID Int fk
}

// Leaders of each Orgs/Provinces
Leaders [icon: crown, color: brown] {
  id Int pk
  enrollID Int fk
  class VARCHAR(50)
  phoneNumber VARCHAR(20) unique
  createdAt TIMESTAMP
}
// Athletics of each Orgs/Provinces
Athletics [icon: runner, color: green] {
  id Int pk
  enrollID Int fk
  class VARCHAR(50)
  uniformSize VARCHAR(4)
  createdAt TIMESTAMP
}

// a collective or types of sports CRUD By super ADMIN
Sports [icon: basketball, color: purple] {
  id Int pk
  name VARCHAR(50)
  createdAt TIMESTAMP
}

// a collective or types of Events/GAMES CRUD By super ADMIN
Events [icon: calendar, color: yellow] {
  id Int pk
  name VARCHAR(80)
  date DATE
  createdAt TIMESTAMP
}

// Leader 
LeaderSports {
  id Int pk
  leaderID Int fk
  sportID Int fk
}

LeaderEvents {
  id Int pk
  leaderID Int fk
  eventID Int fk
}

AthleteSports {
  id Int pk
  athleticsID Int fk
  sportID Int fk
}

AthleteEvents {
  id Int pk
  athleticsID Int fk
  eventID Int fk
}

// Relationships
Enroll.id - Leaders.enrollID          // 1:1 (person is leader)
Enroll.id - Athletics.enrollID        // 1:1 (person is athlete)
Leaders.id > Organization.leaderID    // Many orgs can have same leader? (or reverse if 1 leader per org)
Athletics.id > Organization.athleticsID
Enroll.userID > Users.UserID             // Optional user account link

LeaderSports.leaderID > Leaders.id
LeaderSports.sportID > Sports.id

LeaderEvents.leaderID > Leaders.id
LeaderEvents.eventID > Events.id

AthleteSports.athleticsID > Athletics.id
AthleteSports.sportID > Sports.id

AthleteEvents.athleticsID > Athletics.id
AthleteEvents.eventID > Events.id
